local WhitelistedKeys = {
  "1",
  "2"
}

return function(UserKey)
	local function has_value (table, key)
    	for index, value in ipairs(table) do
   	     if value == key then
    	        return "Found the key!"
	        end
  	  end

  	  return "Did not find the key"
	end
	print(has_value(WhitelistedKeys, UserKey))
end
